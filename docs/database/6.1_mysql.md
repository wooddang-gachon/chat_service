# 카카오톡 데이터베이스 물리 스키마 DDL (5.1_mysql.md)

이 문서는 [6_kakaotalk_db_erd_woo_v2.md](file:///Users/wooddang-mac/Desktop/code/intelliJ/chat_service/docs/database/6_kakaotalk_db_erd_woo_v2.md)에 명세된 모든 테이블 구조와 최적화된 인덱스 설정을 실제 MySQL 데이터베이스에서 즉시 실행하여 구축할 수 있는 물리 DDL(Data Definition Language) 스크립트로 구현한 산출물입니다.

외래키(FK) 참조 오류를 방지하기 위해 부모 테이블이 자식 테이블보다 먼저 생성되도록 순서가 정밀하게 조율되어 있습니다.

---

## 1. DDL SQL 스크립트

```sql
-- -----------------------------------------------------
-- 1. USERS 및 회원 관련 테이블
-- -----------------------------------------------------

-- USERS (회원 마스터)
CREATE TABLE `USERS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '회원 고유 ID',
    `phone_number` VARCHAR(50) NULL COMMENT '국제 규격 전화번호 (Unique)',
    `email` VARCHAR(100) NULL COMMENT '이메일 주소 (Unique)',
    `status` VARCHAR(20) NOT NULL DEFAULT 'ACTIVE' COMMENT 'ACTIVE, BLOCKED, WITHDRAWING',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '가입 일시',
    `status_updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '상태 갱신 일시',
    `deleted_at` TIMESTAMP NULL COMMENT '탈퇴 완료 일시',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_users_phone` (`phone_number`),
    UNIQUE KEY `uk_users_email` (`email`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='회원 기본 테이블';

-- USER_SESSIONS (단일 기기 세션 통제)
CREATE TABLE `USER_SESSIONS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '세션 고유 ID',
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `device_uuid` VARCHAR(100) NOT NULL COMMENT '기기 고유 UUID',
    `session_token` VARCHAR(255) NOT NULL COMMENT '암호화된 세션 토큰 (Unique)',
    `is_active` BOOLEAN NOT NULL DEFAULT TRUE COMMENT '현재 세션 활성화 여부',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '로그인 일시',
    `expires_at` TIMESTAMP NULL COMMENT '만료 예정 일시',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user_sessions_token` (`session_token`),
    CONSTRAINT `fk_user_sessions_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='사용자 로그인 세션';

-- USER_BACKUPS (기기 이전 시 백업 메타데이터)
CREATE TABLE `USER_BACKUPS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '백업 고유 ID',
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `backup_key_hash` VARCHAR(255) NOT NULL COMMENT '비밀번호 해시 값',
    `s3_file_path` VARCHAR(512) NOT NULL COMMENT '백업 파일 S3 위치',
    `expires_at` TIMESTAMP NOT NULL COMMENT '백업 만료 일시 (생성 후 14일)',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '백업 일시',
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_user_backups_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='대화 내역 백업';

-- PROFILES (기본 프로필 정보)
CREATE TABLE `PROFILES` (
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `nickname` VARCHAR(50) NOT NULL COMMENT '프로필 닉네임',
    `profile_image_url` VARCHAR(512) NULL COMMENT '원본 프로필 이미지 S3 주소',
    `thumbnail_url` VARCHAR(512) NULL COMMENT '축소 프로필 이미지 S3 주소',
    `background_image_url` VARCHAR(512) NULL COMMENT '프로필 배경 이미지 S3 주소',
    `status_message` VARCHAR(255) NULL COMMENT '상태 메시지',
    `gender` VARCHAR(10) NULL COMMENT '성별 (MALE, FEMALE, OTHER)',
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시',
    PRIMARY KEY (`user_id`),
    CONSTRAINT `fk_profiles_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='회원 기본 프로필';

-- 생일 및 성별 기반 마케팅 조건 검색 최적화 인덱스
CREATE INDEX `idx_profiles_gender` ON `PROFILES` (`gender`);

-- USER_BIRTHDAYS (생일자 조회 전용 독립 분할 테이블)
CREATE TABLE `USER_BIRTHDAYS` (
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `birth_month` TINYINT NOT NULL COMMENT '생일 월 (1~12)',
    `birth_day` TINYINT NOT NULL COMMENT '생일 일 (1~31)',
    `birth_year` SMALLINT NULL COMMENT '출생 연도 (나이 계산용, 선택)',
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '수정 일시',
    PRIMARY KEY (`user_id`),
    CONSTRAINT `fk_user_birthdays_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='생일자 조회용 인덱스 최적화 테이블';

-- 월/일 기반의 고속 생일자 추출 복합 인덱스
CREATE INDEX `idx_user_birthdays_month_day` ON `USER_BIRTHDAYS` (`birth_month`, `birth_day`);

-- MULTI_PROFILES (멀티프로필)
CREATE TABLE `MULTI_PROFILES` (
    `id` BIGINT AUTO_INCREMENT COMMENT '멀티프로필 고유 ID',
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `nickname` VARCHAR(50) NOT NULL COMMENT '표시 닉네임',
    `profile_image_url` VARCHAR(512) NULL COMMENT '프로필 이미지 S3 주소',
    `background_image_url` VARCHAR(512) NULL COMMENT '배경 이미지 S3 주소',
    `status_message` VARCHAR(255) NULL COMMENT '상태 메시지',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '생성 일시',
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_multi_profiles_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='멀티프로필 설정';

-- FRIENDSHIPS (친구 관계)
CREATE TABLE `FRIENDSHIPS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '관계 식별 고유 ID',
    `user_id` BIGINT NOT NULL COMMENT '기준 사용자 ID',
    `friend_user_id` BIGINT NOT NULL COMMENT '친구 사용자 ID',
    `relationship_type` VARCHAR(20) NOT NULL DEFAULT 'NORMAL' COMMENT 'NORMAL, FAVORITE, HIDDEN, BLOCKED',
    `is_favorite` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '즐겨찾기 여부',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '관계 등록 일시',
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '상태 갱신 일시',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_friendships_user_friend` (`user_id`, `friend_user_id`),
    CONSTRAINT `fk_friendships_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_friendships_friend_id` FOREIGN KEY (`friend_user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='친구 관계 테이블';

-- MULTI_PROFILE_MAPPINGS (멀티프로필 적용 관계 매핑)
CREATE TABLE `MULTI_PROFILE_MAPPINGS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '매핑 고유 ID',
    `multi_profile_id` BIGINT NOT NULL COMMENT '멀티프로필 고유 ID',
    `friendship_id` BIGINT NOT NULL COMMENT '친구 관계 식별 ID',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '매핑 지정 일시',
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_multi_profile_mappings_profile` FOREIGN KEY (`multi_profile_id`) 
        REFERENCES `MULTI_PROFILES` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_multi_profile_mappings_friendship` FOREIGN KEY (`friendship_id`) 
        REFERENCES `FRIENDSHIPS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='멀티프로필-친구 매핑 관계';

-- 내 친구 목록 조회 시 동적 멀티프로필 로딩 가속을 위한 복합 인덱스
CREATE INDEX `idx_multi_profile_mappings_friend_profile` ON `MULTI_PROFILE_MAPPINGS` (`friendship_id`, `multi_profile_id`);


-- -----------------------------------------------------
-- 2. CHAT_ROOMS 및 채팅방 설정 관련 테이블
-- -----------------------------------------------------

-- CHAT_ROOMS (채팅방 공통 마스터)
CREATE TABLE `CHAT_ROOMS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '채팅방 고유 ID',
    `type` VARCHAR(20) NOT NULL COMMENT 'ME, ONE_ON_ONE, GROUP, OPEN_ONE_ON_ONE, OPEN_GROUP',
    `title` VARCHAR(100) NULL COMMENT '채팅방 이름 (단체방용)',
    `room_profile_url` VARCHAR(512) NULL COMMENT '채팅방 프로필 이미지 주소',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '개설 일시',
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='채팅방 메인 마스터';

-- NORMAL_CHAT_ROOMS (일반 채팅방 상세 정보)
CREATE TABLE `NORMAL_CHAT_ROOMS` (
    `chat_room_id` BIGINT NOT NULL COMMENT '채팅방 고유 ID',
    `expires_at` TIMESTAMP NULL COMMENT '채팅방 만료 예정 일시',
    PRIMARY KEY (`chat_room_id`),
    CONSTRAINT `fk_normal_rooms_room_id` FOREIGN KEY (`chat_room_id`) 
        REFERENCES `CHAT_ROOMS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='일반 채팅방 상세';

-- OPEN_CHAT_ROOMS (오픈 채팅방 상세 정보)
CREATE TABLE `OPEN_CHAT_ROOMS` (
    `chat_room_id` BIGINT NOT NULL COMMENT '채팅방 고유 ID',
    `owner_user_id` BIGINT NOT NULL COMMENT '방장 회원 ID',
    `passcode` VARCHAR(50) NULL COMMENT '입장 비밀번호 (선택)',
    `max_members_count` INT NOT NULL DEFAULT 100 COMMENT '입장 제한 정원',
    PRIMARY KEY (`chat_room_id`),
    CONSTRAINT `fk_open_rooms_room_id` FOREIGN KEY (`chat_room_id`) 
        REFERENCES `CHAT_ROOMS` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_open_rooms_owner_id` FOREIGN KEY (`owner_user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='오픈 채팅방 상세';

-- CHAT_ROOM_PARTICIPANTS (채팅방 참여자 관리)
CREATE TABLE `CHAT_ROOM_PARTICIPANTS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '참여자 고유 식별 ID',
    `chat_room_id` BIGINT NOT NULL COMMENT '채팅방 고유 ID',
    `user_id` BIGINT NOT NULL COMMENT '회원 고유 ID',
    `role` VARCHAR(20) NOT NULL DEFAULT 'MEMBER' COMMENT 'OWNER, MANAGER, MEMBER',
    `joined_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '입장 일시',
    `leaved_at` TIMESTAMP NULL COMMENT '퇴장 일시',
    `cleared_at` TIMESTAMP NULL COMMENT '대화 비우기 일시',
    `last_read_message_id` BIGINT NULL COMMENT '마지막으로 읽은 메시지 ID',
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_participants_room_id` FOREIGN KEY (`chat_room_id`) 
        REFERENCES `CHAT_ROOMS` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_participants_user_id` FOREIGN KEY (`user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='채팅방 참여 상태 정보';

-- 특정 유저가 대화방에 재입장하거나 메시지 로딩 시 성능 가속을 위한 인덱스
CREATE UNIQUE INDEX `uk_participants_room_user` ON `CHAT_ROOM_PARTICIPANTS` (`chat_room_id`, `user_id`);

-- CHAT_ROOM_PERSONAL_SETTINGS (참여자 개인 설정 - 수직 분산 적용)
CREATE TABLE `CHAT_ROOM_PERSONAL_SETTINGS` (
    `participant_id` BIGINT NOT NULL COMMENT '참여자 고유 식별 ID',
    `is_notification_on` BOOLEAN NOT NULL DEFAULT TRUE COMMENT '채팅방 알림 활성화 여부',
    `is_pinned` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '상단 고정 여부',
    `pin_order` INT NOT NULL DEFAULT 0 COMMENT '상단 고정 순서 정렬',
    `custom_background_url` VARCHAR(512) NULL COMMENT '채팅방 고유 배경화면 URL',
    `is_input_locked` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '입장 시 입력창 잠금 활성화 여부',
    PRIMARY KEY (`participant_id`),
    CONSTRAINT `fk_personal_settings_part_id` FOREIGN KEY (`participant_id`) 
        REFERENCES `CHAT_ROOM_PARTICIPANTS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='채팅방 참여자별 개인화 옵션';

-- OPEN_CHAT_PARTICIPANT_PROFILES (오픈채팅 참여자 익명 가명 프로필 - 수직 분산 적용)
CREATE TABLE `OPEN_CHAT_PARTICIPANT_PROFILES` (
    `participant_id` BIGINT NOT NULL COMMENT '참여자 고유 식별 ID',
    `open_profile_nickname` VARCHAR(50) NOT NULL COMMENT '가명 프로필 닉네임',
    `open_profile_image_url` VARCHAR(512) NULL COMMENT '가명 프로필 이미지 주소',
    PRIMARY KEY (`participant_id`),
    CONSTRAINT `fk_open_profiles_part_id` FOREIGN KEY (`participant_id`) 
        REFERENCES `CHAT_ROOM_PARTICIPANTS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='오픈채팅방용 익명 프로필 매핑';

-- OPEN_CHAT_BLACKLISTS (오픈채팅 영구 차단 블랙리스트)
CREATE TABLE `OPEN_CHAT_BLACKLISTS` (
    `id` BIGINT AUTO_INCREMENT COMMENT '차단 고유 ID',
    `chat_room_id` BIGINT NOT NULL COMMENT '오픈 채팅방 ID',
    `blocked_user_id` BIGINT NOT NULL COMMENT '차단된 실제 가입자 user_id',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '차단 등록 일시',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_blacklists_room_blocked` (`chat_room_id`, `blocked_user_id`),
    CONSTRAINT `fk_blacklists_room_id` FOREIGN KEY (`chat_room_id`) 
        REFERENCES `CHAT_ROOMS` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_blacklists_blocked_user` FOREIGN KEY (`blocked_user_id`) 
        REFERENCES `USERS` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='오픈채팅방 강퇴 영구 블랙리스트';


-- -----------------------------------------------------
-- 3. MESSAGES 및 메시지 상세 테이블
-- -----------------------------------------------------

-- MESSAGES (메시지 메인 공통 메타)
CREATE TABLE `MESSAGES` (
    `id` BIGINT AUTO_INCREMENT COMMENT '메시지 고유 ID',
    `chat_room_id` BIGINT NOT NULL COMMENT '채팅방 고유 ID',
    `sender_id` BIGINT NOT NULL COMMENT '발신 회원 고유 ID',
    `type` VARCHAR(20) NOT NULL COMMENT 'TEXT, EMOTICON, MINI_EMOTICON, MEDIA, MAP, CONTACT, CALENDAR, SYSTEM',
    `client_message_token` VARCHAR(100) NOT NULL COMMENT '중복 전송(멱등성 보장) 검증용 UUID 토큰',
    `unread_count` INT NOT NULL DEFAULT 0 COMMENT '읽지 않은 사람 수',
    `is_edited` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '수정 여부',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '발송 일시',
    `last_edited_at` TIMESTAMP NULL COMMENT '최종 수정 완료 일시',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_messages_client_token` (`client_message_token`),
    CONSTRAINT `fk_messages_room_id` FOREIGN KEY (`chat_room_id`) 
        REFERENCES `CHAT_ROOMS` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_messages_sender_id` FOREIGN KEY (`sender_id`) 
        REFERENCES `USERS` (`id`) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='채팅 메시지 메인 정보';

-- 대화방 최근 메시지 고속 페이징 조회를 위한 최적화 인덱스
CREATE INDEX `idx_messages_room_id_desc` ON `MESSAGES` (`chat_room_id`, `id` DESC);

-- TEXT_MESSAGES (텍스트 메시지 상세 - 1:1 수직 분할)
CREATE TABLE `TEXT_MESSAGES` (
    `message_id` BIGINT NOT NULL COMMENT '메시지 고유 ID',
    `content` TEXT NOT NULL COMMENT '메시지 본문 텍스트 내용',
    PRIMARY KEY (`message_id`),
    CONSTRAINT `fk_text_messages_msg_id` FOREIGN KEY (`message_id`) 
        REFERENCES `MESSAGES` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='텍스트 메시지 본문';

-- MEDIA_MESSAGES (미디어 및 파일 메시지 상세 - 1:1 수직 분할)
CREATE TABLE `MEDIA_MESSAGES` (
    `message_id` BIGINT NOT NULL COMMENT '메시지 고유 ID',
    `file_type` VARCHAR(20) NOT NULL COMMENT 'PHOTO, VIDEO, FILE',
    `original_s3_path` VARCHAR(512) NOT NULL COMMENT '원본 파일 S3 파일 경로',
    `thumbnail_s3_path` VARCHAR(512) NULL COMMENT '경량 썸네일 이미지 S3 파일 경로',
    `original_expires_at` TIMESTAMP NULL COMMENT '원본 파일 다운로드 만료일',
    `thumbnail_expires_at` TIMESTAMP NULL COMMENT '썸네일 만료일',
    `file_size` BIGINT NOT NULL DEFAULT 0 COMMENT '파일 바이트 용량',
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '전송 일시',
    PRIMARY KEY (`message_id`),
    CONSTRAINT `fk_media_messages_msg_id` FOREIGN KEY (`message_id`) 
        REFERENCES `MESSAGES` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='미디어 메시지 정보';

-- 만료 미디어 파일 자동 제거(배치 성능)를 위한 범위 인덱스
CREATE INDEX `idx_media_original_expires` ON `MEDIA_MESSAGES` (`original_expires_at`);

-- EMOTICON_PACKAGES (이모티콘 패키지 정보)
```
